import 'package:firebase_dynamic_links/firebase_dynamic_links.dart';

Future<String> _createDeepLink({String? file, String? video, String? temple}) async {
  final params = DynamicLinkParameters(
    uriPrefix: 'https://yourapp.page.link',
    link: Uri.parse(
      'https://myapp.com/?'
      '${file != null ? 'file=$file' : ''}'
      '${video != null ? 'video=$video' : ''}'
      '${temple != null ? 'temple=$temple' : ''}'
    ),
    androidParameters: AndroidParameters(packageName: 'com.your.app'),
    iosParameters: IOSParameters(bundleId: 'com.your.app'),
  );
  final shortLink = await FirebaseDynamicLinks.instance.buildShortLink(params);
  return shortLink.shortUrl.toString();
}
